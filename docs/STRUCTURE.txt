```javascript
// Separate builds for member/admin portals
export default defineConfig(({ mode }) => {
  const isAdmin = mode === 'admin';
  return {
    build: {
      outDir: isAdmin ? 'dist-admin' : 'dist',
      rollupOptions: {
        input: isAdmin ? 'admin.html' : 'index.html'
      }
    }
  };
});
Package.json Scripts:
json
{
  "scripts": {
    "dev": "vite",                           // Member dev server
    "dev:admin": "vite --mode admin",        // Admin dev server
    "build": "vite build",                   // Build member portal
    "build:admin": "vite build --mode admin && cd dist-admin && ren admin.html index.html",
    "deploy": "npm run build && firebase deploy --only hosting:nordicrisercom",
    "deploy:admin": "npm run build:admin && firebase deploy --only hosting:nordicrisercom-admin"
  }
}
